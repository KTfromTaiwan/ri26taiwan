// ================================================================ // 把這段貼進 Code.gs（不要刪掉原本的內容，直接附加在最後面） // // 這段讓 GAS Web App 可以接受來自 GitHub Pages 的 fetch 請求 // 並把結果以 JSON 回傳。 // ================================================================ /** * 處理來自 GitHub Pages 的 POST 請求 * Content-Type: text/plain（避免觸發 CORS preflight） * * 支援的 action： * lookupStaff { name } * submitPunch { staffName, pin, type, lat, lng, accuracy } * getMyPunches { staffName, days } */ function doPost(e) { // CORS headers — 讓所有 origin 都能呼叫 const headers = { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Methods': 'POST', 'Access-Control-Allow-Headers': 'Content-Type', }; try { const body = JSON.parse(e.postData.contents); const action = body.action; let result; if (action === 'lookupStaff') { result = lookupStaffByName(body.name); } else if (action === 'submitPunch') { result = submitPunch({ staffName: body.staffName, pin: body.pin, type: body.type, lat: body.lat, lng: body.lng, accuracy: body.accuracy, }); } else if (action === 'getMyPunches') { result = getMyPunches(body.staffName, body.days || 7); } else { result = null; throw new Error('Unknown action: ' + action); } return buildResponse({ result }, headers); } catch (err) { return buildResponse({ error: err.message }, headers); } } /** * 處理瀏覽器的 OPTIONS preflight（某些情況會觸發） */ function doGet(e) { // 如果是普通的 doGet 呼叫（非 preflight），照舊回傳 HTML // （如果你原本的 doGet 已經回傳 HTML，請保留原本的，刪掉這個） return HtmlService.createHtmlOutputFromFile('RI26') .setTitle('Chekcin 打卡') .addMetaTag('viewport', 'width=device-width, initial-scale=1, viewport-fit=cover') .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL); } function buildResponse(data, headers) { const output = ContentService.createTextOutput(JSON.stringify(data)); output.setMimeType(ContentService.MimeType.JSON); // 注意：GAS ContentService 不支援自訂 header， // 但 text/plain body + 部署為「所有人」存取時， // 瀏覽器通常會當作 simple request，不觸發 preflight，可正常讀取。 return output; } // ================================================================ // 部署步驟： // 1. 把上面的 doPost 貼到 Code.gs 最後 // 2. 「部署」→「管理部署」→ 選現有部署 → 「編輯」 // 或「新增部署」→ 類型選「網頁應用程式」 // 執行身份：我（你的 Google 帳號） // 誰可以存取：所有人 // 3. 複製部署網址（https://script.google.com/macros/s/.../exec） // 4. 貼到 checkin-github.html 頂端的 GAS_URL 變數 // ================================================================
